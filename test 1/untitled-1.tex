	\item $c_{ij}^2y_{ij}$ - transportation cost of y from i to j

\begin{align*}
	\min_{x,y} \quad

	\text{s.t.} \quad
	&\sum_{j=1}^mx_{ij}\leq a_{i,1}, \sum_{j=1}^my_{ij}\leq a_{i,2}, \forall i = 1,\dots ,n\\
	&\sum_{i=1}^nx_{ij}\geq b_{j,1}, \sum_{i=1}^ny_{ij}\geq b_{j,2}, \forall j = 1,\dots ,m\\
\end{align*}
